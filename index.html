Animating with CSS

When to use CSS

  • Animations are simple
  • Animate properties that CSS can animate

CSS Transitions - Translation

svg rect {
  transition: transform 1s ease-out;
}

svg:hover rect {
  transform: translateX(100px);
}

CSS Animations - Translation

svg rect {
  animation: move 1s ease-in-out alternate infinite;
}

@keyframes move {
  0% {
    transform: translateX(-50px);
  }

  100% {
    transform: translateX(50px);
  }
}

CSS Transitions - Rotation

svg rect {
  transition: all 1s ease-out;
}

svg:hover rect {
  transform: rotate(360deg);
}

CSS Transitions - Rotation

svg rect {
  transform-origin: 50% 50%;
  transition: all 1s ease-out;
}

svg:hover rect {
  transform: rotate(360deg);
}

Animating with SMIL

When to use SMIL

  • Simple animations
  • Need more control compared to CSS
  • Don't need to support IE or Opera Mini

SMIL – Translation

<animate
  attributeName="x" 
  from="50"
  to="150"
  dur="1s"
  begin="click"
/>

SMIL – Translation (with animateTransform)

<animateTransform 
  attributeName="transform" 
  type="translate"
  from="0, 0"
  to="100, 0"
  dur="1s"
  begin="click"
/>

SMIL – Rotation

<animateTransform 
  attributeName="transform"
  type="rotate"
  from="0"
  to="360"
  dur="1s"
  begin="click"
/>

SMIL – Rotation (fixed)

<animateTransform 
  attributeName="transform"
  type="rotate"
  from="0 100 100"
  to="360 100 100"
  dur="1s"
  repeatCount="indefinite"
  begin="click"
/>

SMIL – More than one animation

<animateTransform 
  type="rotate"
  id="rotate"
  ...
/>

<animate
  attributeName="fill"
  from="#4EDEB9"
  to="#3387F1"
  dur="2s"
  repeatCount='indefinite'
  begin="rotate.begin"
/>

Adding GSAP (TweenMax)

GSAP Syntax

TweenMax.method([elements], duration, destinationProps);

GSAP – Translation

$('rect').on('click', function() {
  TweenMax.to('rect', 1, {
    attr: {
      x: '100'
    },
  })
});

GSAP – Rotation

$('rect').on('click', function() {
  TweenMax.to('rect', 1, {
    rotation: 360
  })
});

GSAP – Rotation (fixed) 

$('rect').on('click', function() {
  TweenMax.to('rect', 1, {
    rotation: 360,
    transformOrigin: '50% 50%'
  })
});

GSAP – Rotation (svgOrigin) 

$('rect').on('click', function() {
  TweenMax.to('rect', 1, {
    rotation: 360,
    svgOrigin: '200 200'
  })
});

  />
</svg>

Line Drawing - Step 1: Split stroke up into dashes

svg path {
  stroke-dasharray: 300
}

Line Drawing - Step 2: Offset the stroke

svg path {
  stroke-dasharray: 300;
  transition: all 2s linear;
}

svg:hover path {
  stroke-dashoffset: -1500;
}

Line Drawing - Step 3: Make strokes longer

svg path {
  stroke-dasharray: 1500;
  transition: all 2s linear;
}

svg:hover path {
  stroke-dashoffset: -1500;
}

Line Drawing - Step 4: Switch offset

svg path {
  stroke-dasharray: 1500;
  stroke-dashoffset: 1500;
  transition: all 2s linear;
}

svg:hover path {
	stroke-dashoffset:0;
}

Line Drawing - Summary

  • Dasharray == path length
  • Dashoffset == path length
  • Easier with Libraries

Snap.svg 

// Initializes Snap
var s = Snap('#button');

// Gets path element 
var elem = s.select('path');

// Start path
var startPath = 'M281,150c0,71.797-59.203,131-131,131S19,221.797,19,150S78.203,19,150,19S281,78.203,281,150z';

// End path

Snap.svg 

$('#button').on('click', function() {
  elem.animate({d: endPath}, 100, mina.easein);
});

Snap.svg 

$('#button').on('click', function() {
  elem.animate({d: endPath}, 100, mina.easein, onComplete);
});

function onComplete () {
  elem.animate({d: startPath}, 800, mina.elastic);
}
